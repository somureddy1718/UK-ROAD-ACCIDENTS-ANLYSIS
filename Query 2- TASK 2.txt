

 Analysis of Car Accident Impacts by Make: Casualties and Vehicles Involved 


 SELECT
    V.Vehicle_Type,
    V.Make,
    FORMAT(SUM(F.Number_of_Casualties), 'N0') AS Total_Casualties,
    FORMAT(SUM(F.Number_of_Vehicles), 'N0') AS Total_Vehicles
FROM 
    Fact_Details F
JOIN
    VehicleDetailsDim V ON F.Vehicle_ID = V.Vehicle_ID
WHERE 
    V.Vehicle_Type = 'Car'
GROUP BY 
    V.Vehicle_Type, V.Make;
